cmake_minimum_required(VERSION 3.20)
project(tsujikiri_compilation_tests CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Fetch pybind11, Lua, and LuaBridge3
include(${CMAKE_SOURCE_DIR}/../cmake/FetchDependencies.cmake)

# -------------------------------------------------------------------------
# Helper macro: add_binding_test(<name> <source_file> <libs...>)
#   Compiles <source_file> with the given libraries; build failure = test failure.
# -------------------------------------------------------------------------
macro(add_binding_test NAME SOURCE)
    add_library(${NAME} SHARED ${SOURCE})
    target_compile_options(${NAME} PRIVATE -Wall -Wextra)
    target_link_libraries(${NAME} PRIVATE ${ARGN})
    # Each test folder's header is on the include path
    target_include_directories(${NAME} PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/${NAME}
    )
endmacro()

# -------------------------------------------------------------------------
# test_compilation — pybind11, LuaBridge3, C API
#
# Source files are generated by tsujikiri and placed next to combined.hpp.
# Run tsujikiri first:
#   tsujikiri -i tests/test_compilation/combined.input.yml -o pybind11 -O tests/test_compilation/pybind11_bindings.cpp
#   tsujikiri -i tests/test_compilation/combined.input.yml -o luabridge3 -O tests/test_compilation/luabridge3_bindings.cpp
# -------------------------------------------------------------------------

set(COMPILATION_DIR ${CMAKE_CURRENT_SOURCE_DIR}/test_compilation)

# --- pybind11 ---
if(EXISTS "${COMPILATION_DIR}/pybind11_bindings.cpp")
    add_library(test_pybind11_bindings MODULE
        ${COMPILATION_DIR}/pybind11_bindings.cpp
    )
    target_include_directories(test_pybind11_bindings PRIVATE ${COMPILATION_DIR})
    target_link_libraries(test_pybind11_bindings PRIVATE pybind11::module)
    set_target_properties(test_pybind11_bindings PROPERTIES
        PREFIX ""
        OUTPUT_NAME "combined"
    )
endif()

# --- LuaBridge3 ---
if(EXISTS "${COMPILATION_DIR}/luabridge3_bindings.cpp")
    add_library(test_luabridge3_bindings STATIC
        ${COMPILATION_DIR}/luabridge3_bindings.cpp
    )
    target_include_directories(test_luabridge3_bindings PRIVATE ${COMPILATION_DIR})
    target_link_libraries(test_luabridge3_bindings PRIVATE LuaBridge3::LuaBridge3)
endif()

# --- C API ---
if(EXISTS "${COMPILATION_DIR}/c_api_bindings.cpp")
    add_library(test_c_api_bindings STATIC
        ${COMPILATION_DIR}/c_api_bindings.cpp
    )
    target_include_directories(test_c_api_bindings PRIVATE ${COMPILATION_DIR})
endif()

# -------------------------------------------------------------------------
# CTest integration (optional)
# -------------------------------------------------------------------------
include(CTest)
